\section{Statistical Estimation}\label{sec:oneone}

\subsection{Uniform Random Variable}\label{sec:urv}
The uniform random variable (RV) is $X \sim U(0, 1)$, and has a probability density function (PDF):

\[ f_{X}(x) = \left\{
	\begin{array}{l l}
		1 & \quad 0 \leq t \leq 1\\
		0 & \quad \text{otherwise}\\
	\end{array}
\right.\]

The expected value of $X$, or \textit{theoretical mean} ($m$) and the standard deviation ($\sigma$) are given by Equation \ref{eq:rvunimu} and \ref{eq:rvunisigma} respectively \cite{IEEEexample:shellCTANpage}.

\resizebox{\textwidth}{!}{%
  \begin{tabular}{ lll }
    \parbox{0.3\textwidth}{\begin{align}
      	m &= \mathbb{E}[X]\notag\\
      	&= \int_{-\infty}^{+\infty} xf_{X}(x) \,dx \notag\\
      	&= \int_{0}^{1}x \,dx \notag\\
      	&= \left[\frac{x^2}{2}\right]_{0}^{1}\notag\\
      	&= \frac{1}{2}\label{eq:rvunimu}
    \end{align}}
    &
    \parbox{0.3\textwidth}{\begin{align*}
    \sigma &= \sqrt{E[X^2] - (E[X])^2}\\
  	E[X^2] &= \int_{-\infty}^{+\infty} x^2 f_{X}(x) \,dx\\
  	&= \int_{0}^{1}x^2 \,dx\\
    &= \left[\frac{x^3}{3}\right]_{0}^{1}\\
  	&= \frac{1}{3}
    \end{align*}}
    &
    \parbox{0.4\textwidth}{\begin{align}
    (E[X])^2 &= \left(\frac{1}{2}\right)^2\notag\\
  	&=\frac{1}{4}\notag\\
  	\sqrt{E[X^2] - (E[X])^2} &= \sqrt{\frac{1}{3} - \frac{1}{4}}\notag\\
  	&=\sqrt{\frac{1}{12}}\notag\\
  	&\approx 0.2887 \label{eq:rvunisigma}
    \end{align}}
  \end{tabular}
  }
